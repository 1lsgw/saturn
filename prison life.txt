local FOV_RADIUS = 150
local FOV_COLOR = Color3.fromRGB(255, 255, 255)
local LINE_COLOR = Color3.fromRGB(255, 0, 0)
local FOV = {
    Radius = FOV_RADIUS,
    Sides = 8,
    Color = FOV_COLOR,
    Lines = {}
}
function FOV:Init()
    for i = 1, self.Sides do
        local l = Drawing.new("Line")
        l.Thickness = 2
        l.Visible = false
        table.insert(self.Lines, l)
    end
end

function FOV:GetPoints(center)
    local pts = {}
    for i = 1, self.Sides do
        local a = (2 * math.pi / self.Sides) * i
        pts[i] = Vector2.new(center.X + math.cos(a) * self.Radius, center.Y + math.sin(a) * self.Radius)
    end
    return pts
end

function FOV:SetVisible(v)
    for _, l in ipairs(self.Lines) do
        l.Visible = v
    end
end

function FOV:Update(center)
    if not ShowVisual or not SilentAim then
        self:SetVisible(false)
        return
    end

    local pts = self:GetPoints(center)
    for i = 1, self.Sides do
        local l = self.Lines[i]
        l.From = pts[i]
        l.To = pts[(i % self.Sides) + 1]
        l.Color = self.Color
        l.Visible = true -- << วงขึ้นเสมอ
    end
end

local function CenterScreen()
    local v = Camera.ViewportSize
    return Vector2.new(v.X / 2, v.Y / 2)
end

local function IsAlive(plr)
    return plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") and
        plr.Character:FindFirstChild("Humanoid") and
        plr.Character.Humanoid.Health > 0
end
local function TeamAllowed(plr)
    if not TeamSelected then
        return false
    end
    return plr.Team and AllowedTeams[plr.Team.Name]
end
local function GetTarget()
    local center = CenterScreen()
    local closest, shortest = nil, FOV_RADIUS
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and IsAlive(plr) and TeamAllowed(plr) then
            local hrp = plr.Character.HumanoidRootPart
            local pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
            if onScreen then
                local dist = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                if dist < shortest then
                    shortest = dist
                    closest = hrp
                end
            end
        end
    end
    return closest
end
RunService.RenderStepped:Connect(
    function()
        local center = CenterScreen()

        FOV.Radius = FOV_RADIUS
        FOV.Color = FOV_COLOR

        if not SilentAim then
            FOV:SetVisible(false)
            AimVisual:Hide()
            CurrentTarget = nil
            return
        end
        FOV:Update(center)
        AimVisual:SetColor(LINE_COLOR)
        if not TeamSelected then
            AimVisual:Hide()
            CurrentTarget = nil
            return
        end
        CurrentTarget = GetTarget()
        if not CurrentTarget then
            AimVisual:Hide()
            return
        end
        local pos, onScreen = Camera:WorldToViewportPoint(CurrentTarget.Position)
        if onScreen then
            AimVisual:Show(center, Vector2.new(pos.X, pos.Y))
        else
            AimVisual:Hide()
        end
    end
)

        local args = {...}
        local method = getnamecallmethod()

                typeof(args[2]) == "Vector3"
         then
            local origin = args[1]
            local dir = (CurrentTarget.Position - origin)
            return __old(self, origin, dir, args[3])
        end
        return __old(self, ...)
    end

                local function patchTool(tool)
                    if not tool:IsA("Tool") then
                        return
                    end
                    pcall(
                        function()
                            tool:SetAttribute("FireRate", FIRE_RATE)
                            tool:SetAttribute("AutoFire", true)
                        end
                    )
                    for _, v in ipairs(tool:GetDescendants()) do
                        if v:IsA("NumberValue") and v.Name:lower():find("fire") then

                            v.Value = true
                        end
                    end
                    for _, v in ipairs(tool:GetDescendants()) do
                        if v:IsA("ModuleScript") then
                            local ok, data = pcall(require, v)
                            if ok and typeof(data) == "table" then
                                if data.FireRate then
                                    data.FireRate = FIRE_RATE
                                end
                                if data.AutoFire ~= nil then
                                    data.AutoFire = true
                                end
                            end
                        end
                    end
                end
                local function onCharacter(char)
                    for _, v in ipairs(char:GetChildren()) do
                        patchTool(v)
                    end
                    char.ChildAdded:Connect(patchTool)
                end
                if player.Character then
                    onCharacter(player.Character)
                end
                player.CharacterAdded:Connect(onCharacter)
                player.Backpack.ChildAdded:Connect(patchTool)

                            obj.PlaybackSpeed = SPEED
                        end
                    end
                end
            end

            local function watchBackpack()
                local backpack = player:WaitForChild("Backpack")

                for _, tool in ipairs(backpack:GetChildren()) do
                    changeSoundsInTool(tool)
                end

                backpack.ChildAdded:Connect(
                    function(tool)
                        task.wait(0.1)
                        changeSoundsInTool(tool)
                    end
                )
            end

            -- ตรวจ Character (ตอนถือ)
            local function watchCharacter(character)
                for _, tool in ipairs(character:GetChildren()) do
                    changeSoundsInTool(tool)
                end

                character.ChildAdded:Connect(
                    function(tool)
                        task.wait(0.1)
                        changeSoundsInTool(tool)
                    end
                )
            end

            -- เริ่มทำงาน
            watchBackpack()

            if player.Character then
                watchCharacter(player.Character)

            -- Services
            local Players = game:GetService("Players")

            local player = Players.LocalPlayer

            -- ปรับค่าได้
            local RGBSpeed = 0.4 -- ความเร็วสีรุ้ง

            local Hue = 0

            -- หา Tool ที่กำลังถือ
            local function GetEquippedTool()
                local char = player.Character
                if not char then
                    return nil
                end

                for _, v in ipairs(char:GetChildren()) do
                    if v:IsA("Tool") then
                        return v
                    end
                end
                return nil
            end

            -- ปรับ Material + RGB ให้ Tool
            local function ApplyToTool(tool)
                if not tool then
                    return
                end

                for _, obj in ipairs(tool:GetDescendants()) do
                    if obj:IsA("BasePart") then
                        obj.Material = Enum.Material.ForceField
                        obj.Color = Color3.fromHSV(Hue, 1, 1)
                    end
                end
            end

            -- Loop RGB
            RunService.RenderStepped:Connect(
                function(dt)
                    Hue = Hue + dt * RGBSpeed
                    if Hue > 1 then
                        Hue = 0
                    end

                    local tool = GetEquippedTool()
                    if tool then
                        ApplyToTool(tool)
                    end
                end
            )

            -- รองรับตอนเกิดใหม่
            player.CharacterAdded:Connect(
                function()
                    Hue = 0
                end
            )
        end
    }

    BillboardGui.Name = "HealthBarESP"
    BillboardGui.Size = UDim2.new(4, 0, 0.5, 0)
    BillboardGui.StudsOffset = Vector3.new(0, 3, 0)
    BillboardGui.AlwaysOnTop = true -- ทำให้เห็นทะลุ
    BillboardGui.ResetOnSpawn = false
    local Frame = Instance.new("Frame")
    Frame.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
    Frame.BorderSizePixel = 0
    Frame.Size = UDim2.new(1, 0, 1, 0)
    Frame.Parent = BillboardGui
    BillboardGui.Parent = game.CoreGui
    ESPs[player] = {HL = hl, GUI = BillboardGui, Frame = Frame}
    local function Update()
        local char = player.Character
        local humanoid = char and char:FindFirstChild("Humanoid")
        local head = char and char:FindFirstChild("Head")
        if ESPEnabled and char and char:FindFirstChild("HumanoidRootPart") then
            hl.Adornee = char
            hl.Enabled = true
            if player.Team then
                local c = player.Team.TeamColor.Color
                hl.FillColor = c
                hl.OutlineColor = c
            else
                hl.FillColor = Color3.new(1, 1, 1)
                hl.OutlineColor = Color3.new(1, 1, 1)
            end
        else
            hl.Enabled = false
        end
        if ESPEnabled and humanoid and head then
            BillboardGui.Adornee = head
            BillboardGui.Enabled = true
            local ratio = math.clamp(humanoid.Health / humanoid.MaxHealth, 0, 1)
            Frame.Size = UDim2.new(ratio, 0, 1, 0)
            if ratio > 0.5 then
                Frame.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
            elseif ratio > 0.2 then
                Frame.BackgroundColor3 = Color3.fromRGB(255, 165, 0)
            else
                Frame.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
            end
        else
            BillboardGui.Enabled = false
        end
    end
    player.CharacterAdded:Connect(Update)
    player:GetPropertyChangedSignal("Team"):Connect(Update)
    game:GetService("RunService").Heartbeat:Connect(Update)
end

        Callback = function()
            -- SERVICES
            local Players = game:GetService("Players")
            local player = Players.LocalPlayer

            -- ตำแหน่งวาร์ป
            local WarpPosition = Vector3.new(-974.60, 108.32, 2049.52)

            -- ฟังก์ชันวาร์ป
            local function WarpCharacter(character)
                local root = character:WaitForChild("HumanoidRootPart", 5)
                if root then
                    root.CFrame = CFrame.new(WarpPosition)
                end
            end

            -- เมื่อเกิดใหม่ → วาร์ปกลับ
            player.CharacterAdded:Connect(
                function(character)
                    task.wait(0.3)
                    WarpCharacter(character)
                end
            )

            -- วาร์ป + บังคับรี
            local function WarpAndRespawn()
                local char = player.Character
                if not char then
                    return
                end

                local hum = char:FindFirstChildOfClass("Humanoid")
                if hum then
                    -- วาร์ปก่อน (กันบางเกม)
                    WarpCharacter(char)

                    task.wait(0.1)

                    -- ฆ่าตัวเพื่อรี
                    hum.Health = 0
                end
            end

            -- เรียกใช้งาน
            WarpAndRespawn()
        end
    }
)
